function doPost(e) { try { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Sheet1"); // مهم جداً: قراءة JSON بشكل صحيح const data = JSON.parse(e.postData.contents); const username = data.username; const proof = data.proof; const wallet = data.wallet; sheet.appendRow([ new Date(), username, proof, wallet ]); return ContentService .createTextOutput(JSON.stringify({ status: "success" })) .setMimeType(ContentService.MimeType.JSON); } catch (error) { return ContentService .createTextOutput(JSON.stringify({ status: "error", message: error.toString() })) .setMimeType(ContentService.MimeType.JSON); } }
